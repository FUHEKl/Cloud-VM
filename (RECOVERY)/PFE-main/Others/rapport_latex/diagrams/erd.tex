  \node[entity] (user) at (0, 0) {

    id: UUID (PK)\\

    firstName: VARCHAR\\

    isRevoked: BOOLEAN
  };

    \textbf{virtual\_machines}\\

    id: UUID (PK)\\

    templateId: VARCHAR\\
    vcpu: INTEGER\\
    ram: INTEGER\\
    disk: INTEGER\\
    status: ENUM\\
    ipAddress: VARCHAR\\
    oneId: INTEGER\\
    createdAt: TIMESTAMP
  };

    maxRAM: INTEGER\\
    maxDisk: INTEGER\\
    price: DECIMAL\\
    isActive: BOOLEAN
  };

    usedRAM: INTEGER\\
    usedDisk: INTEGER
  };

    \textbf{payments}\\
    \hline
    id: UUID (PK)\\
    userId: UUID (FK)\\
    planId: UUID (FK)\\
    amount: DECIMAL\\
    currency: VARCHAR\\
    status: ENUM\\
    stripeSessionId: VARCHAR\\
    createdAt: TIMESTAMP
  };

  \node[entity] (notif) at (0, -8) {
    \textbf{notifications}\\

    id: UUID (PK)\\
    userId: UUID (FK)\\
    title: VARCHAR\\
    message: TEXT\\
    type: ENUM\\
    isRead: BOOLEAN\\
    createdAt: TIMESTAMP
  };
